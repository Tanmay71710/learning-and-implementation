1. Basic Deployment Scenario

Use Case

Overview: Deploy a simple web application with 3 replicas using a Deployment controller. This is the foundational Kubernetes scenario that demonstrates how to deploy and manage containerized applications using Kubernetes Deployments.

What This Use Case Covers:

  • Understanding Kubernetes Deployment controllers and their benefits
  • Creating Deployment manifests for containerized applications
  • Configuring replica counts for high availability
  • Setting up resource requests and limits for proper resource management
  • Implementing health checks (liveness and readiness probes)

When to Use This Scenario:

  • You're new to Kubernetes and want to understand basic deployment concepts
  • You need to deploy a stateless web application
  • You want to ensure high availability with multiple replicas
  • You're setting up basic monitoring and health checks
  • You're learning Kubernetes resource management fundamentals

Implementation Steps:

  1. Create a Deployment manifest specifying the container image and replica count
  2. Configure resource requests and limits to manage CPU and memory usage
  3. Set up liveness probes to detect and restart unhealthy containers
  4. Configure readiness probes to ensure traffic only reaches ready containers
  5. Apply the manifest to your Kubernetes cluster
  6. Verify the deployment is successful and pods are running

Validation Steps:

  • Check deployment status: `kubectl get deployments`
  • Verify pods are running: `kubectl get pods -l app=web-app`
  • Check pod health and restarts: `kubectl describe pod `
  • Verify resource allocation: `kubectl top pods`
  • Test health check endpoints from within the cluster
  • Verify scaling works: `kubectl scale deployment web-app-deployment --replicas=5`

Expected Outcomes:

  • 3 replica pods running the nginx application
  • Automatic pod restarts when containers become unhealthy
  • Proper resource allocation and limits enforced
  • Traffic only routed to ready and healthy containers
  • Deployment can be easily scaled up or down

Deployment Configuration

apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-app-deployment
  labels:
    app: web-app
    environment: production
spec:
  replicas: 3
  selector:
    matchLabels:
      app: web-app
  template:
    metadata:
      labels:
        app: web-app
        environment: production
    spec:
      containers:
      - name: web-app
        image: nginx:1.21
        ports:
        - containerPort: 80
        resources:
          requests:
            memory: "64Mi"
            cpu: "250m"
          limits:
            memory: "128Mi"
            cpu: "500m"
        livenessProbe:
          httpGet:
            path: /
            port: 80
          initialDelaySeconds: 30
          periodSeconds: 10
        readinessProbe:
          httpGet:
            path: /
            port: 80
          initialDelaySeconds: 5
          periodSeconds: 5

Commands to Deploy

kubectl apply -f web-app-deployment.yaml
kubectl get deployments
kubectl get pods -l app=web-app
Tip: Use labels and selectors consistently to ensure proper pod management and service discovery.

Horizontal Pod Autoscaler

apiVersion: autoscaling/v2
kind: HorizontalPodAutoscaler
metadata:
  name: web-app-hpa
spec:
  scaleTargetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: web-app
  minReplicas: 2
  maxReplicas: 10
  metrics:
  - type: Resource
    resource:
      name: cpu
      target:
        type: Utilization
        averageUtilization: 50
  - type: Resource
    resource:
      name: memory
      target:
        type: Utilization
        averageUtilization: 70
  behavior:
    scaleUp:
      stabilizationWindowSeconds: 60
      policies:
      - type: Percent
        value: 50
        periodSeconds: 60
    scaleDown:
      stabilizationWindowSeconds: 300
      policies:
      - type: Percent
        value: 10
        periodSeconds: 60

Vertical Pod Autoscaler

apiVersion: autoscaling.k8s.io/v1
kind: VerticalPodAutoscaler
metadata:
  name: web-app-vpa
spec:
  targetRef:
    apiVersion: apps/v1
    kind: Deployment
    name: web-app
  updatePolicy:
    updateMode: "Auto"
  resourcePolicy:
    containerPolicies:
    - containerName: "*"
      minAllowed:
        cpu: 100m
        memory: 100Mi
      maxAllowed:
        cpu: 1
        memory: 1Gi
      controlledResources: ["cpu", "memory"]

Basic Ingress Rule

apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: web-app-ingress
  annotations:
    nginx.ingress.kubernetes.io/rewrite-target: /

  ingressClassName: nginx

def mutate_pod():
    admission_request = request.json

    # Add default labels if missing
    if 'labels' not in pod['metadata']:
        pod['metadata']['labels'] = {}

        if 'resources' not in container:
            container['resources'] = {

                'limits': {'cpu': '500m', 'memory': '512Mi'}
            }
    
    patch = json.dumps([{

        'value': pod['metadata']['labels']
    }])

        'response': {
            'uid': admission_request['request']['uid'],
            'allowed': True,

            'patchType': 'JSONPatch'
        }
    })

package controllers

import (
    "context"
    "fmt"
    "time"

    appsv1 "k8s.io/api/apps/v1"
    corev1 "k8s.io/api/core/v1"
    "k8s.io/apimachinery/pkg/api/errors"
    metav1 "k8s.io/apimachinery/pkg/apis/meta/v1"
    "k8s.io/apimachinery/pkg/runtime"
    ctrl "sigs.k8s.io/controller-runtime"
    "sigs.k8s.io/controller-runtime/pkg/client"
    "sigs.k8s.io/controller-runtime/pkg/log"

    myappv1 "my-operator/api/v1"
)

type MyAppReconciler struct {
    client.Client
    Scheme *runtime.Scheme
}

func (r *MyAppReconciler) Reconcile(ctx context.Context, req ctrl.Request) (ctrl.Result, error) {
    logger := log.FromContext(ctx)

    myapp := &myappv1.MyApp{}
    err := r.Get(ctx, req.NamespacedName, myapp)
    if err != nil {
        if errors.IsNotFound(err) {
            return ctrl.Result{}, nil
        }
        return ctrl.Result{}, err
    }

    deployment := &appsv1.Deployment{}
    err = r.Get(ctx, req.NamespacedName, deployment)
    if errors.IsNotFound(err) {
        deployment = r.newDeployment(myapp)
        err = r.Create(ctx, deployment)
        if err != nil {
            return ctrl.Result{}, err
        }
        return ctrl.Result{Requeue: true}, nil
    }

    if myapp.Spec.Replicas != *deployment.Spec.Replicas {
        deployment.Spec.Replicas = &myapp.Spec.Replicas
        err = r.Update(ctx, deployment)
        if err != nil {
            return ctrl.Result{}, err
        }
    }

    myapp.Status.AvailableReplicas = deployment.Status.AvailableReplicas
    err = r.Status().Update(ctx, myapp)
    if err != nil {
        return ctrl.Result{}, err
    }

    return ctrl.Result{RequeueAfter: time.Minute * 5}, nil
}

func (r *MyAppReconciler) newDeployment(myapp *myappv1.MyApp) *appsv1.Deployment {
    return &appsv1.Deployment{
        ObjectMeta: metav1.ObjectMeta{
            Name:      myapp.Name,
            Namespace: myapp.Namespace,
        },
        Spec: appsv1.DeploymentSpec{
            Replicas: &myapp.Spec.Replicas,
            Selector: &metav1.LabelSelector{
                MatchLabels: map[string]string{"app": myapp.Name},
            },
            Template: corev1.PodTemplateSpec{
                ObjectMeta: metav1.ObjectMeta{
                    Labels: map[string]string{"app": myapp.Name},
                },
                Spec: corev1.PodSpec{
                    Containers: []corev1.Container{
                        {
                            Name:  myapp.Name,
                            Image: myapp.Spec.Image,
                            Ports: []corev1.ContainerPort{
                                {
                                    ContainerPort: myapp.Spec.Port,
                                },
                            },
                        },
                    },
                },
            },
        },
    }
}

func (r *MyAppReconciler) SetupWithManager(mgr ctrl.Manager) error {
    return ctrl.NewControllerManagedBy(mgr).
        For(&myappv1.MyApp{}).
        Owns(&appsv1.Deployment{}).
        Complete(r)
}
