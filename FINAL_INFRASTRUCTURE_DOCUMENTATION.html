4. Jenkins Pipeline Setup

Step 4.1: Jenkinsfile Creation

Created comprehensive CI/CD pipeline with multiple stages:

pipeline {
    agent any
    
    environment {
        // Artifactory configuration
        ARTIFACTORY_URL = 'http://10.118.244.255:30082'  # Updated for Kubernetes
        ARTIFACTORY_USERNAME = 'admin'
        ARTIFACTORY_PASSWORD = 'password'
        ARTIFACTORY_REPO = 'docker-local'
        
        // Docker configuration
        DOCKER_IMAGE = "script-execution-manager:${BUILD_NUMBER}"
        DOCKER_REPO = "10.118.244.255:30082/${ARTIFACTORY_REPO}/script-execution-manager"
        
        // Application configuration
        FLASK_APP = 'app.py'
        PYTHON_VERSION = '3.9-slim'
    }
    
    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        
        stage('Setup Environment') {
            steps {
                sh """
                    python3 -m venv venv
                    . venv/bin/activate
                    pip install -r requirements.txt
                """
            }
        }
        
        stage('Code Quality') {
            steps {
                sh """
                    . venv/bin/activate
                    echo 'Running code quality checks...'
                """
            }
        }
        
        stage('Build Docker Image') {
            steps {
                sh """
                    docker build -t ${DOCKER_IMAGE} .
                    docker tag ${DOCKER_IMAGE} ${DOCKER_REPO}:${BUILD_NUMBER}
                    docker tag ${DOCKER_IMAGE} ${DOCKER_REPO}:latest
                """
            }
        }
        
        stage('Push to Artifactory') {
            steps {
                sh """
                    echo ${ARTIFACTORY_PASSWORD} | docker login ${ARTIFACTORY_URL} -u ${ARTIFACTORY_USERNAME} --password-stdin
                    docker push ${DOCKER_REPO}:${BUILD_NUMBER}
                    docker push ${DOCKER_REPO}:latest
                """
            }
        }
        
        stage('Test Application') {
            steps {
                sh """
                    . venv/bin/activate
                    echo 'Testing the application...'
                """
            }
        }
        
        stage('Cleanup') {
            steps {
                sh """
                    docker logout ${ARTIFACTORY_URL} || true
                    docker rmi ${DOCKER_IMAGE} || true
                """
            }
        }
    }
    
    post {
        success {
            echo 'Pipeline completed successfully!'
            echo "Docker image: ${DOCKER_REPO}:${BUILD_NUMBER}"
        }
        failure {
            echo 'Pipeline failed! Please check the logs for details.'
        }
        always {
            cleanWs()
        }
    }
}

8. External Access Configuration

Step 8.1: LoadBalancer Issue Analysis

❌ LoadBalancer service not accessible from Windows machine

Problem: LoadBalancer assigned VM IP but wasn't accessible externally

Root Cause: LoadBalancer type requires cloud provider infrastructure; k3s mimics this but doesn't provide actual external load balancing

Step 8.2: Network Configuration

ip addr show
ens192: <BROADCAST,MULTICAST,UP,LOWER_UP>
    inet 10.118.244.255/22 brd 10.118.247.255 scope global ens192

VM Network: 10.118.244.255/22 on interface ens192

Step 8.3: NodePort Solution

Changed service type from LoadBalancer to NodePort:

apiVersion: v1
kind: Service
metadata:
  name: jenkins
  namespace: jenkins
spec:
  type: NodePort  # Changed from LoadBalancer
  selector:
    app: jenkins
  ports:
  - name: http
    port: 8080
    targetPort: 8080
    nodePort: 30080  # External access port
  - name: agent
    port: 50000
    targetPort: 50000
    nodePort: 30050
